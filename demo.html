${meal.strMeal}

Ingredients:

    ${ingredients.map(ingredient => `
  • ${ingredient}
    • `).join('')}
`; targetElement.classList.remove("notFound"); targetElement.innerHTML = html; }); }); } else { html = "Sorry, we didn't find any meal!"; targetElement.classList.add("notFound"); targetElement.innerHTML = html; } }); }
